function filter() {
	let text = document.getElementById("filter").value;
	let items = document.getElementById("items").children;
	for (let i = 0; i < items.length; ++i) {
		if (wildstring.match(text, items[i].innerText)) {
			items[i].classList.add("found");
		}
		else {
			items[i].classList.remove("found");
		}
	}
}

function replace() {
	let text = document.getElementById("replaceText").innerText;
	let replace1 = document.getElementById("replace1").value;
	let replace2 = document.getElementById("replace2").value;
	let out = document.getElementById("replaceOut");
	out.innerText = wildstring.replace(text, [ replace1, replace2 ]);
}
